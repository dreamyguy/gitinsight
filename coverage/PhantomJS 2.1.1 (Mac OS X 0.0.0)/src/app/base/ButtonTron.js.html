import React, {Component} from 'react';
import PropTypes from 'prop-types';
import {connect} from 'react-redux';
import {setChapterAsViewed, setSubChapterAsViewed, setTaskAsViewed} from '../redux/actions/actionsSectors';
 
@connect(store => {
  return {
    viewedPages: store.sectors.viewedPages
  };
})
export class ButtonTron extends Component {
  constructor() {
    super();
    this.handleNavigate = this.navigate.bind(this);
  }
  navigate(path) {
    const {sectorId, chapterId, subChapterId, taskId} = this.props.router.params;
    if (chapterId && !subChapterId) {
      // eslint-disable-next-line react/prop-types
      this.props.dispatch(setChapterAsViewed(path, sectorId, chapterId));
    }
    if (chapterId && subChapterId && !taskId) {
      // eslint-disable-next-line react/prop-types
      this.props.dispatch(setSubChapterAsViewed(path, sectorId, chapterId, subChapterId));
    }
    if (chapterId && subChapterId && taskId) {
      // eslint-disable-next-line react/prop-types
      this.props.dispatch(setTaskAsViewed(path, sectorId, chapterId, subChapterId, taskId));
    }
    // eslint-disable-next-line react/prop-types
    this.props.router.push(path);
  }
  render() {
    return (
      <button
        className="button button-tron" onClick={() => {
          this.handleNavigate(this.props.url);
        }}>
        <div className="button-tron-title">{this.props.title}</div>
      </button>
    );
  }
}
 
ButtonTron.propTypes = {
  url: PropTypes.string,
  title: PropTypes.string,
  router: PropTypes.object.isRequired
};
