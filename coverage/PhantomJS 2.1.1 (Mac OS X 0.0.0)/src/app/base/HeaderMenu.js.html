import _ from 'lodash';
import React, {Component} from 'react';
import PropTypes from 'prop-types';
import {connect} from 'react-redux';
 
import {fetchHeaderMenu, headerMenuToggler} from '../redux/actions/actionsFrontend';
import {ButtonEpic} from './ButtonEpic';
import './HeaderMenu.scss';
 
@connect(store => {
  return {
    menuState: store.frontend.menuState,
    menuItems: store.frontend.menuItems
  };
})
 
export class HeaderMenu extends Component {
  constructor() {
    super();
    // this.handleSetSelectedSector = this.setSelectedSector.bind(this);
    this.handleNavigate = this.navigate.bind(this);
    // this.handleSetSelectedSector = this.setSelectedSector.bind(this);
    this.handleToggleHeaderMenu = this.toggleHeaderMenu.bind(this);
  }
  componentWillMount() {
    const {menuItems} = this.props;
    // When accessing the page directly for the first time, the 'menuItems' object
    // is defined, but is empty. In those cases we should fetch the data. This
    // check also prevents a new fetch in case 'menuItems' already has data.
    if (menuItems && Object.keys(menuItems).length === 0 && menuItems.constructor === Object) {
      // eslint-disable-next-line react/prop-types
      this.props.dispatch(fetchHeaderMenu());
    }
  }
  navigate(path) {
    // eslint-disable-next-line react/prop-types
    this.props.router.push(path);
  }
  toggleHeaderMenu(state) {
    if (state === 'collapsed') {
      // eslint-disable-next-line react/prop-types
      this.props.dispatch(headerMenuToggler('expanded'));
    } else if (state === 'expanded') {
      // eslint-disable-next-line react/prop-types
      this.props.dispatch(headerMenuToggler('collapsed'));
    }
  }
  renderHeaderMenu() {
    return _.map(this.props.menuItems, item => {
      return (
        <li key={item.Id}>
          <ButtonEpic url={item.Url} title={item.Name} router={this.props.router}/>
        </li>
      );
    });
  }
  render() {
    const {menuState} = this.props;
    return (
      <div className="header-menu">
        <button
          className="header-menu-toggler" onClick={() => {
            this.handleToggleHeaderMenu(menuState);
          }}>
          <p className={`nav-mobile-icon ${menuState}`}/>
        </button>
        <div className={`header-menu-list ${menuState}`}>
          <ul>
            {this.renderHeaderMenu()}
          </ul>
        </div>
      </div>
    );
  }
}
 
HeaderMenu.propTypes = {
  router: PropTypes.object.isRequired,
  menuState: PropTypes.string,
  menuItems: PropTypes.object
};
