/* eslint react/prop-types: "off" */
 
import React, {Component} from 'react';
import {connect} from 'react-redux';
 
import {fetchUser} from '../../redux/actions/actionsUser';
import {fetchSessions} from '../../redux/actions/actionsSessions';
 
@connect(store => {
  return {
    user: store.user.user,
    userFetched: store.user.fetched,
    sessions: store.session.sessions
  };
})
export class User extends Component {
  constructor() {
    super();
    this.handleClick = this.fetchSessions.bind(this);
  }
  componentWillMount() {
    this.props.dispatch(fetchUser());
  }
  fetchSessions() {
    this.props.dispatch(fetchSessions());
  }
  render() {
    const {user, sessions} = this.props;
 
    if (!sessions.length) {
      return <button onClick={this.handleClick}>load sessions</button>;
    }
 
    const mappedSessions = sessions.map((session, i) => <li key={i}>{session._id}</li>);
 
    return (
      <div>
        <h1>{user.id}</h1>
        <ul>{mappedSessions}</ul>
      </div>
    );
  }
}
