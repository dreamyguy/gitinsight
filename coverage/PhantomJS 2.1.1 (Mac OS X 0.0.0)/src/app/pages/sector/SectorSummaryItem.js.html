/* eslint "react/no-danger": "off" */
// @todo: if it can be helped, get rid of this 'dangerouslySetInnerHTML' helper.
 
import React, {Component} from 'react';
import PropTypes from 'prop-types';
 
export class SectorSummaryItem extends Component {
  renderQuestion(content) {
    const title = content.Title;
    const description = content.Description;
    const question = content.Question;
    if (question) {
      const questionText = question.Text;
      let questionClass = 'bs-callout-success'; // answered
      if (questionText) {
        let questionAnswer = question.Answer;
        if (questionAnswer === null) {
          questionAnswer = 'Ubesvart';
          questionClass = 'bs-callout-danger'; // unanswered
        }
        return (
          <div>
            <div className={`bs-callout ${questionClass}`}>
              <h3>{title}</h3>
              <span dangerouslySetInnerHTML={{__html: description}}/>
              <p><strong>Svar:</strong> {questionAnswer}</p>
            </div>
          </div>
        );
      }
    }
  }
  render() {
    const {sectorSummaryItem} = this.props;
    return (
      <div>
        {this.renderQuestion(sectorSummaryItem)}
      </div>
    );
  }
}
 
SectorSummaryItem.propTypes = {
  sectorSummaryItem: PropTypes.object.isRequired
};
