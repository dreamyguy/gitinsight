import _ from 'lodash';
import React, {Component} from 'react';
import {connect} from 'react-redux';
import PropTypes from 'prop-types';
 
import {fetchSectors} from '../../redux/actions/actionsSectors';
import {Title} from '../../base/Title';
import {Description} from '../../base/Description';
import {SectorSummaryItem} from './SectorSummaryItem';
 
@connect(store => {
  return {
    sectors: store.sectors.sectors
  };
})
export class SectorSummary extends Component {
  componentWillMount() {
    const {sectors} = this.props;
    // When accessing the page directly for the first time, the 'sectors' object
    // is defined, but is empty. In those cases we should fetch the data. This
    // check also prevents a new fetch in case 'sectors' already has data.
    if (sectors && Object.keys(sectors).length === 0 && sectors.constructor === Object) {
      // eslint-disable-next-line react/prop-types
      this.props.dispatch(fetchSectors());
    }
  }
  renderSectorSummaryItem(pages) {
    return _.map(pages, item => {
      return (
        <SectorSummaryItem key={item.PageId} sectorSummaryItem={item}/>
      );
    });
  }
  render() {
    const sectorId = this.props.sectorId || this.props.router.params.sectorId;
    const sector = this.props.sectors[sectorId];
    if (!sector) {
      return <div>Laster...</div>;
    }
    const title = sector.Title;
    const description = sector.Description;
    const pages = sector.Pages;
    return (
      <div>
        <Title title={title}/>
        <Description description={description}/>
        {this.renderSectorSummaryItem(pages)}
      </div>
    );
  }
}
 
SectorSummary.propTypes = {
  router: PropTypes.object.isRequired,
  sectors: PropTypes.array,
  sectorId: PropTypes.string
};
