/* eslint react/jsx-no-bind: 0 */
/* eslint react/jsx-closing-bracket-location: 0 */
 
import _ from 'lodash';
import React, {Component} from 'react';
import PropTypes from 'prop-types';
import {connect} from 'react-redux';
 
import logMessage from '../../helpers/logMessage';
import {fetchSectors, setAnswerSubChapter, setSubChapterAsViewed} from '../../redux/actions/actionsSectors';
import {Description} from '../../base/Description';
import {TextReader} from '../../base/TextReader';
import {PageTitle} from '../../base/PageTitle';
import {ButtonNav} from '../../base/ButtonNav';
import {ButtonBack} from '../../base/ButtonBack';
import './SectorChapter.scss';
 
@connect(store => {
  return {
    sectors: store.sectors.sectors,
    viewedPages: store.sectors.viewedPages,
    currentAnswer: store.sectors.currentAnswer
  };
})
export class SectorSubChapter extends Component {
  constructor() {
    super();
    this.handleNavigate = this.navigate.bind(this);
    this.handleSetAnswerSubChapter = this.getAndSetAnswerSubChapter.bind(this);
  }
  componentWillMount() {
    const {sectors} = this.props;
    const {sectorId, chapterId} = this.props.router.params;
    const {pathname} = this.props.router.location;
    // When accessing the chapter directly for the first time, the 'sectors' object
    // is defined, but is empty. In those cases we should fetch the data. This
    // check also prevents a new fetch in case 'sectors' already has data.
    if (sectors && Object.keys(sectors).length === 0 && sectors.constructor === Object) {
      // eslint-disable-next-line react/prop-types
      this.props.dispatch(fetchSectors(pathname, sectorId, chapterId));
    }
  }
  navigate(path) {
    const {sectorId, chapterId, subChapterId} = this.props.router.params;
    // eslint-disable-next-line react/prop-types
    this.props.dispatch(setSubChapterAsViewed(path, sectorId, chapterId, subChapterId));
    // eslint-disable-next-line react/prop-types
    this.props.router.push(path);
  }
  getAndSetAnswerSubChapter(answer) {
    const {sectorId, chapterId, subChapterId} = this.props.router.params;
    // eslint-disable-next-line react/prop-types
    this.props.dispatch(setAnswerSubChapter(answer, sectorId, chapterId, subChapterId));
  }
  getFirstTask() {
    const {sectorId, chapterId, subChapterId} = this.props.router.params;
    return <ButtonNav url={`/bransje/${sectorId}/side/${chapterId}/relevant/${subChapterId}/tiltak/1`} title="Fortsett" router={this.props.router} navType="dive"/>;
  }
  getNextSubChapter(propOwner) {
    const {sectorId, chapterId, subChapterId} = this.props.router.params;
    const nextPageId = parseInt(subChapterId, 10) + 1;
    if (!propOwner[nextPageId]) {
      return logMessage('NO NEXT PAGE');
    }
    return <ButtonNav url={`/bransje/${sectorId}/side/${chapterId}/relevant/${nextPageId}`} title="Neste" router={this.props.router} navType="next"/>;
  }
  getPreviousTask(propOwner) {
    const {sectorId, chapterId, subChapterId} = this.props.router.params;
    const previousPageId = parseInt(subChapterId, 10) - 1;
    if (!propOwner[previousPageId]) {
      return logMessage('NO PREVIOUS PAGE');
    }
    return <ButtonBack url={`/bransje/${sectorId}/side/${chapterId}/relevant/${previousPageId}`} title="Forrige" router={this.props.router}/>;
  }
  getCurrentSectionPage() {
    const {sectorId, chapterId, subChapterId} = this.props.router.params;
    const currentPageId = parseInt(subChapterId, 10);
    let firstPage = false;
    if (currentPageId === 1) {
      firstPage = true;
    }
    if (firstPage) {
      return <ButtonBack url={`/bransje/${sectorId}/side/${chapterId}`} title="Forrige" router={this.props.router}/>;
    }
  }
  renderAnswerButtonsSubChapter(answer) {
    let ansY = null;
    let ansN = null;
    if (answer === 'Ja') {
      ansY = 'answer-yes';
      ansN = '';
    } else if (answer === 'Nei') {
      ansY = '';
      ansN = 'answer-no';
    }
    return (
      <div className="mb-3">
        <ul className="answer-selectors">
          <li>
            <button
              className={`answer ${ansY}`} onClick={() => {
                // eslint-disable-next-line react/prop-types
                this.handleSetAnswerSubChapter('Ja');
              }}
            />
            <span className="answer-label">Ja</span>
          </li>
          <li>
            <button
              className={`answer ${ansN}`} onClick={() => {
                // eslint-disable-next-line react/prop-types
                this.handleSetAnswerSubChapter('Nei');
              }}
            />
            <span className="answer-label">Nei</span>
          </li>
        </ul>
      </div>
    );
  }
  renderQuestionSubChapter(subchapter) {
    return (
      <div>
        <Description description={subchapter.Question.Text}/>
        {this.renderAnswerButtonsSubChapter(subchapter.Question.Answer)}
      </div>
    );
  }
  renderSubChapter(chapter) {
    const {subChapterId} = this.props.router.params;
    const subchapter = chapter.Pages[subChapterId];
    return (
      <div>
        <PageTitle title={chapter.Title}/>
        <div className="content">
          <Description description={subchapter.Description}/>
          {this.renderQuestionSubChapter(subchapter)}
        </div>
      </div>
    );
  }
  renderNavigationSubChapters(subchapters) {
    const {sectors} = this.props;
    const {sectorId, chapterId, subChapterId} = this.props.router.params;
    const answer = sectors[sectorId].Pages[chapterId].Pages[subChapterId].Question.Answer;
    let nextClass;
    if (answer !== null) {
      nextClass = answer.toLowerCase();
    }
    return (
      <div className="sector-navigation">
        <div>
          {this.getCurrentSectionPage(subchapters)}
          {this.getPreviousTask(subchapters)}
        </div>
        <div className={`dive-${nextClass}`}>
          {this.getNextSubChapter(subchapters)}
          {this.getFirstTask()}
        </div>
      </div>
    );
  }
  render() {
    const {chapterId, sectorId} = this.props.router.params;
    const sector = this.props.sectors[sectorId];
    if (!sector) {
      return <div>Laster...</div>;
    }
    const title = sector.Title;
    const chapters = sector.Pages;
    const chapter = chapters[chapterId];
    const subchapters = chapter.Pages;
    return (
      <div className="sector-page">
        <TextReader text={title}/>
        {this.renderSubChapter(chapter)}
        <div className="content">
          {this.renderNavigationSubChapters(subchapters)}
        </div>
      </div>
    );
  }
}
 
SectorSubChapter.propTypes = {
  sectors: PropTypes.array,
  router: PropTypes.object.isRequired
};
