import _ from 'lodash';
import React, {Component} from 'react';
import {connect} from 'react-redux';
import PropTypes from 'prop-types';
 
import {fetchSectors} from '../../redux/actions/actionsSectors';
import {HTMLRender} from '../../base/HTMLRender';
import {ButtonTron} from '../../base/ButtonTron';
import './Sector.scss';
 
@connect(store => {
  return {
    sectors: store.sectors.sectors
  };
})
export class Sector extends Component {
  constructor() {
    super();
    this.handleNavigate = this.navigate.bind(this);
  }
  componentWillMount() {
    const {sectors} = this.props;
    // When accessing the page directly for the first time, the 'sectors' object
    // is defined, but is empty. In those cases we should fetch the data. This
    // check also prevents a new fetch in case 'sectors' already has data.
    if (sectors && Object.keys(sectors).length === 0 && sectors.constructor === Object) {
      // eslint-disable-next-line react/prop-types
      this.props.dispatch(fetchSectors());
    }
  }
  navigate(path) {
    // eslint-disable-next-line react/prop-types
    this.props.router.push(path);
  }
  render() {
    const sectorId = this.props.router.params.sectorId;
    const sector = this.props.sectors[sectorId];
    if (!sector) {
      return <div>Laster...</div>;
    }
    const title = sector.Title;
    const description = sector.Description;
    const cat = 'https://www.petfinder.com/wp-content/uploads/2012/11/91615172-find-a-lump-on-cats-skin-632x475.jpg';
    return (
      <div className="sector">
        <div className="img sector-img" style={{backgroundImage: `url(${cat})`}}>
          <div className="sector-img-title">
            {title}
          </div>
          <div className="sector-img-description">
            <HTMLRender html={description}/>
          </div>
        </div>
        <ButtonTron title={'Start'} url={`/bransje/${sectorId}/side/1`} router={this.props.router}/>
      </div>
    );
  }
}
 
Sector.propTypes = {
  router: PropTypes.object.isRequired,
  sectors: PropTypes.array,
  sectorId: PropTypes.string
};
