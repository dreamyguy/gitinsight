import _ from 'lodash';
import React, {Component} from 'react';
import PropTypes from 'prop-types';
import {connect} from 'react-redux';
 
import {fetchSectors, setSelectedSector} from '../../redux/actions/actionsSectors';
import {Title} from '../../base/Title';
import {Description} from '../../base/Description';
 
import './Sectors.scss';
 
@connect(store => {
  return {
    sectors: store.sectors.sectors,
    selectedSector: store.sectors.selectedSector
  };
})
export class Sectors extends Component {
  constructor() {
    super();
    this.handleSetSelectedSector = this.setSelectedSector.bind(this);
    this.handleNavigate = this.navigate.bind(this);
  }
  componentWillMount() {
    const {sectors} = this.props;
    // When accessing the page directly for the first time, the 'sectors' object
    // is defined, but is empty. In those cases we should fetch the data. This
    // check also prevents a new fetch in case 'sectors' already has data.
    if (sectors && Object.keys(sectors).length === 0 && sectors.constructor === Object) {
      // eslint-disable-next-line react/prop-types
      this.props.dispatch(fetchSectors());
    }
  }
  setSelectedSector(id) {
    // eslint-disable-next-line react/prop-types
    this.props.dispatch(setSelectedSector(id));
  }
  navigate(path) {
    // eslint-disable-next-line react/prop-types
    this.props.router.push(path);
  }
  renderSectors() {
    return _.map(this.props.sectors, sector => {
      return (
        <li key={sector.Sector}>
          <button
            className="button button-epic" onClick={() => {
              this.handleSetSelectedSector(sector.Sector);
              this.handleNavigate(`/bransje/${sector.Sector}`);
            }}>
            <div className="button-epic-grid">
              <div className="button-epic-title button-epic-grid-item">{sector.Title}</div>
              <div className="button-epic-arrow button-epic-grid-item">
                <div className="icon">&#x3031;</div>
              </div>
            </div>
          </button>
        </li>
      );
    });
  }
  render() {
    return (
      <div className="content">
        <Title title="Velg bransje"/>
        <Description description="Her må du velge en bransje slik at du kan begynne tilsvarende risikovurderingen."/>
        <ul className="sectors-list">
          {this.renderSectors()}
        </ul>
      </div>
    );
  }
}
 
Sectors.propTypes = {
  sectors: PropTypes.array
};
