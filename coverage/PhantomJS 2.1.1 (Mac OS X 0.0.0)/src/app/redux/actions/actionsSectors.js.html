import axios from 'axios';
import {lastUpdated} from './actionsFrontend';
 
export function fetchSectors(path, sectorId, chapterId) {
  const SERVER = 'localhost';
  // const SERVER = '172.16.205.21';
  const API_ROOT = `http://${SERVER}:7000/api`;
  return dispatch => {
    dispatch({type: 'FETCH_SECTORS'});
 
    axios.get(`${API_ROOT}/sectors`)
      .then(response => {
        dispatch({type: 'FETCH_SECTORS_FULFILLED', payload: response.data});
        dispatch(setChapterAsViewed(path, sectorId, chapterId));
        dispatch(lastUpdated());
      })
      .catch(err => {
        dispatch({type: 'FETCH_SECTORS_REJECTED', payload: err});
      });
  };
}
