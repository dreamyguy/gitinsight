import _ from 'lodash';
 
export default function reducer(state = {
  currentAnswer: null,
  sectors: {},
  selectedSector: [],
  viewedPages: [],
  fetchingSectors: false,
  fetchedSectors: false,
  errorSectors: null
}, action) {
  switch (action.type) {
    case 'FETCH_SECTORS':
      {
        return {
          ...state,
          fetchingSectors: true
        };
      }
    case 'FETCH_SECTORS_REJECTED':
      {
        return {
          ...state,
          fetchingSectors: false,
          errorSectors: action.payload
        };
      }
    case 'FETCH_SECTORS_FULFILLED':
      {
        return {
          ...state,
          fetchingSectors: false,
          fetchedSectors: true,
          sectors: _.mapKeys(action.payload, 'Sector')
        };
      }
    case 'SET_SELECTED_SECTOR':
      {
        return {
          ...state,
          selectedSector: action.payload.id
        };
      }
    case 'SET_CHAPTER_AS_VIEWED':
      {
        const newViewedPages = [...state.viewedPages];
        if (newViewedPages.indexOf(action.payload.path) === -1) {
          newViewedPages.push(action.payload.path);
        } else {
          // If we land here it means this page has already been visited.
          // Change 'sectors[sectorId].Pages[chapterId].View' from 'No' to 'Yes'
          return {
            ...state,
            sectors: {
              ...state.sectors,
              [action.payload.sectorId]: {
                ...state.sectors[action.payload.sectorId],
                Pages: {
                  ...state.sectors[action.payload.sectorId].Pages,
                  [action.payload.chapterId]: {
                    ...state.sectors[action.payload.sectorId].Pages[action.payload.chapterId],
                    Viewed: 'Yes'
                  }
                }
              }
            }
          };
        }
        return {
          ...state,
          viewedPages: newViewedPages
        };
      }
    case 'SET_SUBCHAPTER_AS_VIEWED':
      {
        const newViewedPages = [...state.viewedPages];
        if (newViewedPages.indexOf(action.payload.path) === -1) {
          newViewedPages.push(action.payload.path);
        } else {
          // If we land here it means this page has already been visited.
          // Change 'sectors[sectorId].Pages[chapterId].View' from 'No' to 'Yes'
          return {
            ...state,
            sectors: {
              ...state.sectors,
              [action.payload.sectorId]: {
                ...state.sectors[action.payload.sectorId],
                Pages: {
                  ...state.sectors[action.payload.sectorId].Pages,
                  [action.payload.chapterId]: {
                    ...state.sectors[action.payload.sectorId].Pages[action.payload.chapterId],
                    Pages: {
                      ...state.sectors[action.payload.sectorId].Pages[action.payload.chapterId].Pages,
                      [action.payload.subChapterId]: {
                        ...state.sectors[action.payload.sectorId].Pages[action.payload.chapterId].Pages[action.payload.subChapterId],
                        Viewed: 'Yes'
                      }
                    }
                  }
                }
              }
            }
          };
        }
        return {
          ...state,
          viewedPages: newViewedPages
        };
      }
    case 'SET_TASK_AS_VIEWED':
      {
        const newViewedPages = [...state.viewedPages];
        if (newViewedPages.indexOf(action.payload.path) === -1) {
          newViewedPages.push(action.payload.path);
        } else {
          // If we land here it means this page has already been visited.
          // Change 'sectors[sectorId].Pages[chapterId].View' from 'No' to 'Yes'
          return {
            ...state,
            sectors: {
              ...state.sectors,
              [action.payload.sectorId]: {
                ...state.sectors[action.payload.sectorId],
                Pages: {
                  ...state.sectors[action.payload.sectorId].Pages,
                  [action.payload.chapterId]: {
                    ...state.sectors[action.payload.sectorId].Pages[action.payload.chapterId],
                    Pages: {
                      ...state.sectors[action.payload.sectorId].Pages[action.payload.chapterId].Pages,
                      [action.payload.subChapterId]: {
                        ...state.sectors[action.payload.sectorId].Pages[action.payload.chapterId].Pages[action.payload.subChapterId],
                        Pages: {
                          ...state.sectors[action.payload.sectorId].Pages[action.payload.chapterId].Pages[action.payload.subChapterId].Pages,
                          [action.payload.taskId]: {
                            ...state.sectors[action.payload.sectorId].Pages[action.payload.chapterId].Pages[action.payload.subChapterId].Pages[action.payload.taskId],
                            Viewed: 'Yes'
                          }
                        }
                      }
                    }
                  }
                }
              }
            }
          };
        }
        return {
          ...state,
          viewedPages: newViewedPages
        };
      }
    case 'SET_ANSWER_CHAPTER':
      {
        return {
          ...state,
          sectors: {
            ...state.sectors,
            [action.payload.sectorId]: {
              ...state.sectors[action.payload.sectorId],
              Pages: {
                ...state.sectors[action.payload.sectorId].Pages,
                [action.payload.chapterId]: {
                  ...state.sectors[action.payload.sectorId].Pages[action.payload.chapterId],
                  Question: {
                    ...state.sectors[action.payload.sectorId].Pages[action.payload.chapterId].Question,
                    Answer: action.payload.answer
                  }
                }
              }
            }
          },
          currentAnswer: action.payload.answer
        };
      }
    case 'SET_ANSWER_SUBCHAPTER':
      {
        return {
          ...state,
          sectors: {
            ...state.sectors,
            [action.payload.sectorId]: {
              ...state.sectors[action.payload.sectorId],
              Pages: {
                ...state.sectors[action.payload.sectorId].Pages,
                [action.payload.chapterId]: {
                  ...state.sectors[action.payload.sectorId].Pages[action.payload.chapterId],
                  Pages: {
                    ...state.sectors[action.payload.sectorId].Pages[action.payload.chapterId].Pages,
                    [action.payload.subChapterId]: {
                      ...state.sectors[action.payload.sectorId].Pages[action.payload.chapterId].Pages[action.payload.subChapterId],
                      Question: {
                        ...state.sectors[action.payload.sectorId].Pages[action.payload.chapterId].Pages[action.payload.subChapterId].Question,
                        Answer: action.payload.answer
                      }
                    }
                  }
                }
              }
            }
          },
          currentAnswer: action.payload.answer
        };
      }
    case 'SET_ANSWER_TASK':
      {
        return {
          ...state,
          sectors: {
            ...state.sectors,
            [action.payload.sectorId]: {
              ...state.sectors[action.payload.sectorId],
              Pages: {
                ...state.sectors[action.payload.sectorId].Pages,
                [action.payload.chapterId]: {
                  ...state.sectors[action.payload.sectorId].Pages[action.payload.chapterId],
                  Pages: {
                    ...state.sectors[action.payload.sectorId].Pages[action.payload.chapterId].Pages,
                    [action.payload.subChapterId]: {
                      ...state.sectors[action.payload.sectorId].Pages[action.payload.chapterId].Pages[action.payload.subChapterId],
                      Pages: {
                        ...state.sectors[action.payload.sectorId].Pages[action.payload.chapterId].Pages[action.payload.subChapterId].Pages,
                        [action.payload.taskId]: {
                          ...state.sectors[action.payload.sectorId].Pages[action.payload.chapterId].Pages[action.payload.subChapterId].Pages[action.payload.taskId],
                          Question: {
                            ...state.sectors[action.payload.sectorId].Pages[action.payload.chapterId].Pages[action.payload.subChapterId].Pages[action.payload.taskId].Question,
                            Answer: action.payload.answer
                          }
                        }
                      }
                    }
                  }
                }
              }
            }
          },
          currentAnswer: action.payload.answer
        };
      }
    case 'UPDATE_SECTOR':
      {
        const {id} = action.payload;
        const newSectors = [...state.sectors];
        const sectorToUpdate = newSectors.findIndex(sector => sector.id === id);
        newSectors[sectorToUpdate] = action.payload;
        return {
          ...state,
          sectors: newSectors
        };
      }
    default:
      return state;
  }
}
